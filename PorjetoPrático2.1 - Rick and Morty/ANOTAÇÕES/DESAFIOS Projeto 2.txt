async function separarPorTemporada() {
  const urlBase = 'https://rickandmortyapi.com/api/episode';
  let url = urlBase;
  let todosOsEpisodios = [];

  // Passo 1: Buscar todos os episódios (API é paginada)
  while (url) {
    const resposta = await fetch(url);
    const dados = await resposta.json();
    todosOsEpisodios = todosOsEpisodios.concat(dados.results);
    url = dados.info.next; // vai para próxima página
  }

  // Passo 2: Separar por temporada com base no código "S01E01"
  const temporadas = {};

  todosOsEpisodios.forEach(episodio => {
    const codTemporada = episodio.episode.slice(1, 3); // "S01E01" → "01"
    if (!temporadas[codTemporada]) {
      temporadas[codTemporada] = [];
    }
    temporadas[codTemporada].push(episodio);
  });

  // Exibir no console por temporada
  Object.entries(temporadas).forEach(([numTemporada, episodios]) => {
    console.log(`📺 Temporada ${parseInt(numTemporada)}:`);
    episodios.forEach(ep => {
      console.log(`   ${ep.episode} - ${ep.name}`);
    });
  });

  return temporadas; // você pode usar isso para mostrar na tela
}

separarPorTemporada();
